% Definition Box
\usepackage{tcolorbox}
\tcbuselibrary{skins, breakable}

% Info Box (Blue)
\newtcolorbox{infobox}[1][]{
    colback=blue!5!white,
    colframe=blue!75!black,
    fonttitle=\bfseries,
    title=#1,
    breakable
}
